import numpy as np
import pandas as pd
from scipy.cluster.hierarchy import linkage, dendrogram, fcluster
import matplotlib.pyplot as plt


# 2. Perform Hierarchical Clustering using Ward's method
Z_ward_euclidean = linkage(customers_data_for_algorithm, method='ward')

# 3. Plot the dendrogram
plt.figure(figsize=(15, 6))

# Subplot 1: Ward's method with Euclidean distance
plt.subplot(1, 2, 1)
dendrogram(Z_ward_euclidean)
plt.title("Dendrogram: Ward's Method (Euclidean Distance)")
plt.xlabel("Data Points")
plt.ylabel("Distance")

plt.tight_layout()
plt.show()

# 4. Determine the optimal number of clusters and assign them
# For this example, let's assume we want to form 3 clusters.
num_clusters = 3
cluster_labels = fcluster(Z_ward_euclidean, num_clusters, criterion='maxclust')

# 5. Add the cluster labels to the original DataFrame
customers_data_for_algorithm['Cluster'] = cluster_labels
