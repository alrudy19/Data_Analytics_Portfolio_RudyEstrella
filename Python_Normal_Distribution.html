prob_less_than_value1 = stats.norm.cdf(value1, loc=mean, scale=std_dev)

prob_less_than_value2 = stats.norm.cdf(value2, loc=mean, scale=std_dev)

prob_between = prob_less_than_value1 - prob_less_than_value2

result_percent = round(prob_between * 100, 2)
prob_value1_percent = round(prob_less_than_value1 * 100, 2)
prob_value2_percent = round(prob_less_than_value2 * 100, 2)

x = np.linspace(mean - 4 * std_dev, mean + 4 * std_dev, 1000)
y = stats.norm.pdf(x, loc=mean, scale=std_dev)

plt.figure(figsize=(10, 6))
plt.plot(x, y, color='blue', label='Normal Distribution (PDF)')

x_fill_value1 = np.linspace(min(x), value1, 1000)
y_fill_value1 = stats.norm.pdf(x_fill_value1, loc=mean, scale=std_dev)

plt.fill_between(x_fill_value1, y_fill_value1, color='skyblue', alpha=0.8,
                 label=f'P(X <= {value1}) = {prob_value1_percent}%')

x_fill_value2 = np.linspace(min(x), value2, 1000)
y_fill_value2 = stats.norm.pdf(x_fill_value2, loc=mean, scale=std_dev)
plt.fill_between(x_fill_value2, y_fill_value2, color='lightgray', alpha=0.9,
                 label=f'P(X <= {value2}) = {prob_value2_percent}%')

plt.text(mean, 0.008, f'P({value2} < X <= {value1}) = {result_percent}%', fontsize=15,
         ha='center', va='center', bbox=dict(facecolor='white', alpha=0.5))

plt.title(f'Normal Distribution with Mean={mean} and Std Dev={std_dev}', fontsize=16)
plt.xlabel('Value', fontsize=12)
plt.ylabel('Probability Density', fontsize=12)
plt.legend(loc='upper left')
plt.grid(True, linestyle='--', alpha=0.6)
plt.show()
